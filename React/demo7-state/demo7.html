Demo7: state

<script type="text/jsx">
    var Hello = React.createClass({
        getInitialState: function() {
            return {
                flag: true
            };
        },
        doClick: function() {
            this.setState({
                flag: !this.state.flag
            });
        },
        render: function() {
            return(
                <h2 onClick={ this.doClick }>
                    Hello, { this.state.flag.toString() }
                </h2>
            );
        }
    });
    ReactDOM.render(
        <Hello/>,
        document.getElementById('container')
    );
</script>
