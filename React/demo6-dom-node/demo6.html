Demo6: get dom node by ref

<script type="text/jsx">
    var Hello = React.createClass({
        componentDidMount: function() {
            var that = this;
            setTimeout(function(){
                that.refs.test.innerHTML = 'Hello, reset DOM Node innerHTML';
            }, 2000);
        },
        render: function() {
            return (
                <div>
                    <h2 ref="test">
                        Hello, get dom node!
                    </h2>
                </div>
            );
        }
    });
            
    ReactDOM.render(
        <Hello />,
        document.getElementById('container')
    );
</script>
