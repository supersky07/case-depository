Demo11: render with data

<script type="text/jsx">

    var tempData = [

        {

            age: 11,

            name: 'LiLei',

            sex: 'male'

        },

        {

            age: 10,

            name: 'HanMeimei',

            sex: 'femail'

        }

    ];

    var Item = React.createClass({

        render: function() {

            return (

                <div>

                    <label>{ this.props.labelTxt }：</label>

                    <span>{ this.props.content }</span>

                </div>

            );

        }

    });

    var Card = React.createClass({

        render: function() {

            var info = this.props.info;

            return (

                <div style={{ borderBottom: 'dotted 1px #000', marginTop: '10px' }}>

                    <Item labelTxt='姓名' content={ info.name } />

                    <Item labelTxt='年龄' content={ info.age } />

                    <Item labelTxt='性别' content={ info.sex } />

                </div>

            );

        }

    });

    var Hello = React.createClass({

        render: function() {

            var dataList = this.props.data;

            var list = [];

            for (var i = 0, len = dataList.length; i < len; i++) {

                list.push(<Card info={ dataList[i] } />);

            }

            return <div>{ list }</div>;

        }

    });

    ReactDOM.render(

        <Hello data={ tempData } />,

        document.getElementById('container')

    );

</script>
