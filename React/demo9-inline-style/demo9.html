Demo9: inline styles

<script type="text/jsx">
    var h1Style = {
        color: 'red',
        backgroundColor: '#ccc',
        WebkitTransition: 'all',
        msTransition: 'all'
    };

    var Hello = React.createClass({
        render: function() {
            return <h2 style={ h1Style } className="test">Hello, inline styles！</h2>;
        }
    });
            
    ReactDOM.render(
        <Hello />,
        document.getElementById('container')
    );
</script>
